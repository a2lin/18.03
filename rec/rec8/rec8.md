mr^2 + br + k &= 0 \\\

\\\
{-1\over2} &= {-b\over 2m} \\\
b &= m\\\
\\\
{\sqrt{b^2-4mk}\over 2m} &= 3i \\\
{b^2-4mk} &= -36m^2 \\\
m^2 - 4mk &= -36m^2 \\\
-4mk &= -37m^2 \\\
k &= {37\over 4}m \\\
\end{align}
